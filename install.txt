npm install -g create-react-app => instalação react
npx create-react-app nome-do-projeto => criação dos projetos
npm start => inicia o server

cd tab => pastas dentro da pasta que você estiver

//Angular
npm install -g @angular/cli => instalação angular
ng new nome-do-projeto ou  npx ng new nome-do-projeto => criação dos projetos
ng server = > inicia o server
ng g c header => cria uma página header
ng g c footer => cria uma página footer
